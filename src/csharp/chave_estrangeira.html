using System;

using System.ComponentModel.DataAnnotations;

using System.ComponentModel.DataAnnotations.Schema;

using System.Web.Mvc;

namespace SistemaAssinaturaDigital.Models

{

[Table("SGSTF.ASD_PARAMETRO")]

public class Parametro

{

[Key]

[Column("PRT_IN_SEQUENCIA")]

public int Codigo { get; set; }


//relacionamento de chave estrangeira

//o entity framework sabe que é um relacionamento n para 1

public virtual Empresa Empresa { get; set; }


[Required(ErrorMessage = "O Codigo da Empresa é obrigatorio")]

[Display(Name = "Codigo da Empresa")]

[ForeignKey(nameof(Empresa)), Column("EMP_IN_CODIGO")]

public int CodigoEmpresa { get; set; }


[Required(ErrorMessage = "Data da Vigência é Obrigatorio")]

[Display(Name = "Data da Vigênica")]

[Column("PRT_DT_VIGENCIA")]

[DisplayFormat(ApplyFormatInEditMode = true, DataFormatString = "{0:yyyy-MM-dd}")]//aparecer no campo o valor da data

[DataType(DataType.Date)]//Formatação para não salvar a hora

public DateTime DataVigencia { get; set; }


[Column("USU_ST_INCLUSAO")]

public string UsuarioInclusao { get; set; }


[Column("USU_DT_INCLUSAO")]

public DateTime? DataInclusao { get; set; }


[Column("USU_ST_ALTERACAO")]

public string UsuarioAlteracao { get; set; }


[Column("USU_DT_ALTERACAO")]

public DateTime? DataAlteracao { get; set; }


[NotMapped]//não vai mapear

[Display(Name = "Empresas")]

public SelectList EmpresasList { get; set; }

//selectlist não precisar criar uma viewModel>/

}

}//Fim da Class

Action Post Salvar

Action Post Salvar

[HttpPost]

[ValidateAntiForgeryToken]

public ActionResult Salvar(Parametro parametro)

{

try

{

if (!ModelState.IsValid) return View(parametro);


var UsuarioLogado = User.Identity.Name;

if (parametro.Codigo != 0)

{

db.Entry(parametro).State = EntityState.Modified;

parametro.UsuarioAlteracao = UsuarioLogado;

parametro.DataAlteracao = DateTime.Now;

AlertHandler.AddNotification(AlertNotification.Success(Mensagens.SucessoEditarRegistro));

}

else

{

db.Entry(parametro).State = EntityState.Added;

parametro.UsuarioInclusao = UsuarioLogado;

parametro.DataInclusao = DateTime.Now;

AlertHandler.AddNotification(AlertNotification.Success(Mensagens.SucessoSalvarRegistro));

}


db.SaveChanges();


return RedirectToAction("Index");

}

catch (Exception ex)

{

AlertHandler.AddNotification(AlertNotification.Success(Mensagens.ErroSalvarUsuario + " " + ex.Message));

return View("Cadastrar", parametro);

}

}

Action Post Deletar

Action Post Deletar

[HttpDelete]

public JsonResult Deletar(int Codigo)

{

try

{

var parametro = db.Parametros.Find(Codigo);

if (parametro == null)

{

return Json(new { Descricao = Mensagens.ErroExcluirItem + "Item não encontrado", Status = Mensagens.StatusErro });

}

else

{

db.Parametros.Remove(parametro);

db.SaveChanges();

return Json(new { Descricao = Mensagens.SucessoExcluirRegistro, Status = Mensagens.StatusSucesso });

}

}

catch (Exception ex)

{

return Json(new { Descricao = Mensagens.ErroExcluirItem + ex.Message, Status = Mensagens.StatusErro });

}

}
